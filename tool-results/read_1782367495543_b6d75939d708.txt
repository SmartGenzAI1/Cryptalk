    50→const QUICK_REACTIONS = ['👍', '❤️', '🔥', '😂', '😮', '🎉', '👏', '🙏']
    51→
    52→interface MessageItemProps {
    53→  message: MessageWithSender
    54→  isOwn: boolean
    55→  isFirstInGroup: boolean
    56→  isLastInGroup: boolean
    57→  showAvatar: boolean
    58→}
    59→
    60→function MessageItemImpl({ message, isOwn, isFirstInGroup, isLastInGroup }: MessageItemProps) {
    61→  const currentUser = useChatStore((s) => s.currentUser)
    62→  const activeChat = useChatStore((s) => s.activeChat)
    63→  const updateMessage = useChatStore((s) => s.updateMessage)
    64→  const removeMessage = useChatStore((s) => s.removeMessage)
    65→  const [editing, setEditing] = useState(false)
    66→  const [editText, setEditText] = useState(message.content)
    67→  // F1: initialize from the server value so previously-starred messages show
    68→  // the star indicator on mount (was always `false`). We use the React 19
    69→  // "adjust state during render" pattern (not a useEffect) to sync when
    70→  // `message.starred` changes from elsewhere — avoids the lint rule against
    71→  // calling setState synchronously inside an effect.
    72→  const [starred, setStarred] = useState(() => !!message.starred)
    73→  const [prevStarred, setPrevStarred] = useState(message.starred)
    74→  if (message.starred !== prevStarred) {
    75→    setPrevStarred(message.starred)
    76→    setStarred(!!message.starred)
    77→  }
    78→  const [showHoverBar, setShowHoverBar] = useState(false)
    79→  const [showQuickReact, setShowQuickReact] = useState(false)
    80→  const [forwardOpen, setForwardOpen] = useState(false)
    81→  // voice playback
    82→  const [playing, setPlaying] = useState(false)
    83→  const [progress, setProgress] = useState(0)
    84→  const playTimer = useRef<ReturnType<typeof setInterval> | null>(null)
    85→
    86→  // Attachment resolution state — for image / file / voice messages the
    87→  // `content` field may be an encrypted URL (Supabase path), an encrypted
    88→  // data URL (dev fallback), or the literal "[delivered]" placeholder after
    89→  // the server wipes content post-delivery. We resolve it asynchronously.
    90→  const [attachment, setAttachment] = useState<{
    91→    status: 'loading' | 'ready' | 'delivered' | 'error'
    92→    dataUrl: string | null
    93→  }>({ status: 'loading', dataUrl: null })
    94→
    95→  const isDeleted = !!message.deletedAt
    96→  const isSystem = message.type === 'system'
    97→  const isVoice = message.type === 'voice'
    98→  const isSticker = message.type === 'sticker'
    99→  const isImage = message.type === 'image'
   100→  const isFile = message.type === 'file'
   101→
   102→  // Resolve attachment content for image / file / voice messages.
   103→  // Deps are intentionally limited to message.id + message.content (stable
   104→  // across re-renders of the memoized component) plus the chat type which
   105→  // is needed for E2EE decryption.
   106→  const chatType = activeChat?.type || 'direct'
   107→  useEffect(() => {
   108→    if (!isImage && !isFile && !isVoice) return
   109→    let cancelled = false
   110→
   111→    async function resolve() {
   112→      const raw = message.content
   113→      if (!raw) {
   114→        if (!cancelled) setAttachment({ status: 'error', dataUrl: null })
   115→        return
   116→      }
   117→      // Fast-path: server wiped content after delivery confirmation
   118→      if (raw === '[delivered]') {
   119→        if (!cancelled) setAttachment({ status: 'delivered', dataUrl: null })
   120→        return
   121→      }
   122→      try {
   123→        let resolved = raw
   124→        // If content isn't already a URL or data URL, it's an encrypted
   125→        // ciphertext JSON string — decrypt it first.
   126→        if (
   127→          !resolved.startsWith('http://') &&
   128→          !resolved.startsWith('https://') &&
   129→          !resolved.startsWith('data:')
   130→        ) {
   131→          const { decryptMessageForChat } = await import('@/lib/e2ee')
   132→          resolved = await decryptMessageForChat(resolved, message.chatId, chatType)
   133→        }
   134→
   135→        if (cancelled) return
   136→
   137→        if (resolved === '[delivered]') {
   138→          setAttachment({ status: 'delivered', dataUrl: null })
   139→          return
   140→        }
   141→        if (resolved.startsWith('http://') || resolved.startsWith('https://')) {
   142→          // Supabase-stored attachment — fetch ciphertext bytes & decrypt
   143→          const dataUrl = await fetchAndDecryptAttachment(resolved, message.chatId, chatType)
   144→          if (!cancelled) setAttachment({ status: 'ready', dataUrl })
   145→        } else if (resolved.startsWith('data:')) {
   146→          // Dev fallback — content is already the decrypted data URL
   147→          if (!cancelled) setAttachment({ status: 'ready', dataUrl: resolved })
   148→        } else {
   149→          // Legacy plaintext (no E2EE) — treat as data URL if it looks like one,
   150→          // otherwise mark as error.
   151→          if (!cancelled) setAttachment({ status: 'ready', dataUrl: resolved })
   152→        }
   153→      } catch {
   154→        if (!cancelled) setAttachment({ status: 'error', dataUrl: null })
   155→      }
   156→    }
   157→
   158→    resolve()
   159→    return () => { cancelled = true }
   160→  }, [message.id, message.content, message.chatId, chatType, isImage, isFile, isVoice])
   161→
   162→  // reaction grouped by emoji
   163→  const reactionGroups = message.reactions.reduce<Record<string, { count: number; mine: boolean }>>((acc, r) => {
   164→    if (!acc[r.emoji]) acc[r.emoji] = { count: 0, mine: false }
   165→    acc[r.emoji].count++
   166→    if (r.user.id === currentUser?.id) acc[r.emoji].mine = true
   167→    return acc
   168→  }, {})
   169→
   170→  async function handleReact(emoji: string) {
   171→    try {
   172→      const data = await toggleReaction(message.chatId, message.id, emoji)
   173→      getSocket()?.emit('reaction', {
   174→        chatId: message.chatId,
   175→        messageId: message.id,
   176→        emoji,
   177→        userId: currentUser?.id,
   178→        added: data.added,
   179→      })
   180→      // locally refresh this message reactions
   181→      const refreshed = await apiGet<{ messages: any[] }>(`/api/${message.chatId}/messages?limit=200`)
   182→      if (refreshed.messages) {
   183→        const m = refreshed.messages.find((x: any) => x.id === message.id)
   184→        if (m) updateMessage(message.chatId, m)
   185→      }
   186→      setShowQuickReact(false)
   187→    } catch (e) {
   188→      console.error(e)
   189→    }
   190→  }
   191→
   192→  async function handleEdit() {
   193→    if (!editText.trim()) return
   194→    try {
   195→      const data = await apiPatch<{ message: any }>(`/api/${message.chatId}/messages?messageId=${message.id}`, {
   196→        content: editText.trim(),
   197→      })
   198→      if (data.message) {
   199→        updateMessage(message.chatId, data.message)
   200→        getSocket()?.emit('message-update', { chatId: message.chatId, message: data.message, action: 'edit' })
   201→      }
   202→      setEditing(false)
   203→    } catch (e) {
   204→      console.error(e)
   205→    }
   206→  }
   207→
   208→  async function handleDelete(forEveryone: boolean = false) {
   209→    try {
   210→      await apiDelete(`/api/${message.chatId}/messages?messageId=${message.id}${forEveryone ? '&forEveryone=true' : ''}`)
   211→      removeMessage(message.chatId, message.id)
   212→      getSocket()?.emit('message-update', {
   213→        chatId: message.chatId,
   214→        message: { ...message, deletedAt: new Date().toISOString(), content: '🗑️ Message deleted' },
   215→        action: 'delete',
   216→      })
   217→      toast.success(forEveryone ? 'Deleted for everyone' : 'Message deleted')
   218→    } catch (e) {
   219→      console.error(e)
   220→    }
   221→  }
   222→
   223→  async function handleStar() {
   224→    try {
   225→      const data = await toggleStar(message.chatId, message.id)
   226→      setStarred(data.starred)
   227→      toast.success(data.starred ? 'Added to starred' : 'Removed from starred')
   228→    } catch (e) {
   229→      console.error(e)
   230→    }
   231→  }
   232→
   233→  function handleCopy() {
   234→    navigator.clipboard.writeText(message.content)
   235→    toast.success('Copied to clipboard')
   236→  }
   237→
   238→  function startReply() {
   239→    window.dispatchEvent(new CustomEvent('zc-reply', { detail: message }))
   240→  }
   241→
   242→  const audioRef = useRef<HTMLAudioElement | null>(null)
   243→
   244→  function togglePlay() {
   245→    if (playing) {
   246→      setPlaying(false)
   247→      audioRef.current?.pause()
   248→      return
   249→    }
   250→
   251→    const audioContent = attachment.status === 'ready' ? attachment.dataUrl : null
   252→
   253→    if (!audioContent) {
   254→      if (attachment.status === 'delivered') {
   255→        toast.error('Voice message no longer available')
   256→      } else if (attachment.status === 'error') {
   257→        toast.error('Could not load voice message')
   258→      } else {
   259→        toast.error('Voice message is still loading…')
   260→      }
   261→      return
   262→    }
   263→
   264→    if (!audioContent.startsWith('data:audio')) {
   265→      // legacy voice message (no actual audio) — simulate playback
   266→      setPlaying(true)
   267→      const duration = message.duration || 5
   268→      const step = 100 / (duration * 10)
   269→      playTimer.current = setInterval(() => {
   270→        setProgress((p) => {
   271→          const next = p + step
   272→          if (next >= 100) {
   273→            setPlaying(false)
   274→            if (playTimer.current) clearInterval(playTimer.current)
   275→            return 0
   276→          }
   277→          return next
   278→        })
   279→      }, 100)
   280→      return
   281→    }
   282→
   283→    const audio = new Audio(audioContent)
   284→    audioRef.current = audio
   285→    audio.onended = () => {
   286→      setPlaying(false)
   287→      setProgress(0)
   288→    }
   289→    audio.ontimeupdate = () => {
   290→      if (audio.duration) {
   291→        setProgress((audio.currentTime / audio.duration) * 100)
   292→      }
   293→    }
   294→    audio.play()
   295→    setPlaying(true)
   296→  }
   297→
   298→  useEffect(() => {
   299→    return () => {
   300→      audioRef.current?.pause()
   301→      if (playTimer.current) clearInterval(playTimer.current)
   302→    }
   303→  }, [])
   304→
   305→  if (isSystem) {
   306→    return (
   307→      <div className="flex justify-center my-2">
   308→        <span className="px-3 py-1 rounded-full bg-background/80 backdrop-blur text-xs text-muted-foreground border">
   309→          {message.content}
   310→        </span>
   311→      </div>
   312→    )
   313→  }
   314→
   315→  const HoverActions = (
   316→    <AnimatePresence>
   317→      {showHoverBar && !editing && !isDeleted && (
   318→        <motion.div
   319→          initial={{ opacity: 0, scale: 0.85 }}
   320→          animate={{ opacity: 1, scale: 1 }}
   321→          exit={{ opacity: 0, scale: 0.85 }}
   322→          transition={{ duration: 0.15 }}
   323→          className={cn(
   324→            'absolute top-1/2 -translate-y-1/2 flex items-center gap-0.5 bg-background/95 backdrop-blur rounded-full shadow-lg border p-0.5 z-20',
   325→            isOwn ? '-left-1' : '-right-1'
   326→          )}
   327→          onMouseLeave={() => setShowHoverBar(false)}
   328→        >
   329→          <button onClick={() => setShowQuickReact((v) => !v)} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="React">
   330→            <Smile className="h-4 w-4" />
   331→          </button>
   332→          <button onClick={startReply} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Reply">
   333→            <Reply className="h-4 w-4" />
   334→          </button>
   335→          <button onClick={() => setForwardOpen(true)} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Forward">
   336→            <Forward className="h-4 w-4" />
   337→          </button>
   338→          <button onClick={handleStar} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Star">
   339→            <Star className={cn('h-4 w-4', starred && 'fill-amber-400 text-amber-400')} />
   340→          </button>
   341→          {isOwn && (
   342→            <button onClick={() => { setEditing(true); setEditText(message.content) }} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Edit">
   343→              <Pencil className="h-4 w-4" />
   344→            </button>
   345→          )}
   346→        </motion.div>
   347→      )}
   348→    </AnimatePresence>
   349→  )
   350→
   351→  return (
   352→    <>
   353→      <ContextMenu>
   354→        <ContextMenuTrigger asChild>
   355→          <div
   356→            className={cn('group relative flex items-end gap-2 px-1', isOwn ? 'justify-end' : 'justify-start', isLastInGroup ? 'mb-2' : 'mb-0.5')}
   357→            onMouseEnter={() => setShowHoverBar(true)}
   358→            onMouseLeave={() => { setShowHoverBar(false); setShowQuickReact(false) }}
   359→          >
   360→            {!isOwn && (
   361→              <div className="w-8 shrink-0">
   362→                {isLastInGroup && (
   363→                  <ChatAvatar emoji={message.sender.avatarEmoji} color={message.sender.avatarColor} size="sm" userId={message.sender.id} />
   364→                )}
   365→              </div>
   366→            )}
   367→
   368→            <div className={cn('relative max-w-[85%] sm:max-w-[65%] min-w-0 flex flex-col', isOwn ? 'items-end' : 'items-start')}>
   369→              {!isOwn && isFirstInGroup && activeChat?.type !== 'direct' && activeChat?.type !== 'saved' && (
   370→                <span className="text-xs font-bold ml-1 mb-0.5 text-primary">
   371→                  {message.sender.name}
   372→                </span>
   373→              )}
   374→
   375→              {HoverActions}
   376→
   377→              {/* Quick react popover */}
   378→              <AnimatePresence>
   379→                {showQuickReact && (
   380→                  <motion.div
   381→                    initial={{ opacity: 0, y: 8, scale: 0.9 }}
   382→                    animate={{ opacity: 1, y: 0, scale: 1 }}
   383→                    exit={{ opacity: 0, y: 8, scale: 0.9 }}
   384→                    transition={{ type: 'spring', stiffness: 400, damping: 25 }}
   385→                    className={cn(
   386→                      'absolute -top-11 bg-background/95 backdrop-blur rounded-full shadow-lg border p-1 flex items-center gap-0.5 z-30',
   387→                      isOwn ? 'right-0' : 'left-0'
   388→                    )}
   389→                  >
   390→                    {QUICK_REACTIONS.map((e) => (
   391→                      <button
   392→                        key={e}
   393→                        onClick={() => handleReact(e)}
   394→                        className="h-8 w-8 rounded-full hover:bg-accent hover:scale-125 transition-all flex items-center justify-center text-lg"
   395→                      >
   396→                        {e}
   397→                      </button>
   398→                    ))}
   399→                  </motion.div>
   400→                )}
   401→              </AnimatePresence>
   402→
   403→              <motion.div
   404→                layout
   405→                initial={{ opacity: 0, y: 6, scale: 0.97 }}
   406→                animate={{ opacity: 1, y: 0, scale: 1 }}
   407→                transition={{ type: 'spring', stiffness: 380, damping: 28 }}
   408→                className={cn(
   409→                  'relative rounded-2xl px-3.5 py-2 shadow-sm',
   410→                  isOwn
   411→                    ? 'bg-gradient-to-br from-primary to-primary text-primary-foreground rounded-br-md'
   412→                    : 'bg-background border rounded-bl-md',
   413→                  isDeleted && 'opacity-60 italic',
   414→                  starred && 'ring-1 ring-amber-400/50'
   415→                )}
   416→              >
   417→                {/* Reply preview */}
   418→                {message.replyTo && (
   419→                  <div className={cn('mb-1.5 pl-2 border-l-2 text-xs', isOwn ? 'border-white/50' : 'border-primary')}>
   420→                    <div className={cn('font-semibold', isOwn ? 'text-white/90' : 'text-primary')}>
   421→                      {message.replyTo.senderName}
   422→                    </div>
   423→                    <div className={cn('truncate max-w-[200px]', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   424→                      {message.replyTo.content}
   425→                    </div>
   426→                  </div>
   427→                )}
   428→
   429→                {editing ? (
   430→                  <div className="min-w-[200px]">
   431→                    <textarea
   432→                      value={editText}
   433→                      onChange={(e) => setEditText(e.target.value)}
   434→                      className={cn(
   435→                        'w-full bg-transparent outline-none resize-none text-sm',
   436→                        isOwn ? 'placeholder-white/60' : ''
   437→                      )}
   438→                      rows={2}
   439→                      autoFocus
   440→                    />
   441→                    <div className="flex justify-end gap-1 mt-1">
   442→                      <Button size="sm" variant="ghost" className="h-7 text-xs" onClick={() => { setEditing(false); setEditText(message.content) }}>
   443→                        Cancel
   444→                      </Button>
   445→                      <Button size="sm" className="h-7 text-xs bg-emerald-600 hover:bg-emerald-700" onClick={handleEdit}>
   446→                        Save
   447→                      </Button>
   448→                    </div>
   449→                  </div>
   450→                ) : isSticker ? (
   451→                  isAnimatedSticker(message.content) ? (
   452→                    <Suspense fallback={<span className="text-6xl leading-none">⭐</span>}>
   453→                      <AnimatedStickerDisplay name={message.content} size={140} />
   454→                    </Suspense>
   455→                  ) : isLegacyEmoji(message.content) ? (
   456→                    <span className="text-6xl leading-none">{message.content}</span>
   457→                  ) : (
   458→                    <img
   459→                      src={stickerIconUrl(message.content)}
   460→                      alt={message.content}
   461→                      width={128}
   462→                      height={128}
   463→                      loading="lazy"
   464→                      className="object-contain"
   465→                    />
   466→                  )
   467→                ) : isImage ? (
   468→                  attachment.status === 'delivered' ? (
   469→                    <AttachmentPlaceholder text="Image no longer available (delivered & wiped)" />
   470→                  ) : attachment.status === 'loading' ? (
   471→                    <AttachmentLoading label="Loading image…" />
   472→                  ) : attachment.status === 'error' ? (
   473→                    <AttachmentPlaceholder text="Failed to load image" isError />
   474→                  ) : attachment.dataUrl && attachment.dataUrl.startsWith('data:image') ? (
   475→                    <img
   476→                      src={attachment.dataUrl}
   477→                      alt="shared"
   478→                      loading="lazy"
   479→                      className="rounded-lg max-w-[280px] max-h-[280px] object-contain cursor-pointer"
   480→                      onClick={() => attachment.dataUrl && window.open(attachment.dataUrl, '_blank')}
   481→                    />
   482→                  ) : (
   483→                    <AttachmentPlaceholder text="Unsupported image format" isError />
   484→                  )
   485→                ) : isFile ? (
   486→                  attachment.status === 'delivered' ? (
   487→                    <AttachmentPlaceholder text="File no longer available (delivered & wiped)" />
   488→                  ) : attachment.status === 'loading' ? (
   489→                    <AttachmentLoading label="Loading file…" />
   490→                  ) : attachment.status === 'error' ? (
   491→                    <AttachmentPlaceholder text="Failed to load file" isError />
   492→                  ) : (
   493→                    <div className="flex items-center gap-2 min-w-[200px]">
   494→                      <a
   495→                        href={attachment.dataUrl || '#'}
   496→                        download={attachment.dataUrl ? 'file' : undefined}
   497→                        className="flex items-center gap-2 px-3 py-2 rounded-lg bg-black/10 hover:bg-black/20 transition-colors"
   498→                      >
   499→                        <FileIcon className="h-5 w-5" />
   500→                        <span className="text-sm">Download file</span>
   501→                      </a>
   502→                    </div>
   503→                  )
   504→                ) : isVoice ? (
   505→                  attachment.status === 'delivered' ? (
   506→                    <AttachmentPlaceholder text="Voice message no longer available (delivered & wiped)" />
   507→                  ) : attachment.status === 'error' ? (
   508→                    <AttachmentPlaceholder text="Failed to load voice message" isError />
   509→                  ) : (
   510→                    <VoiceBubble
   511→                      duration={message.duration || 5}
   512→                      playing={playing}
   513→                      progress={progress}
   514→                      onToggle={togglePlay}
   515→                      isOwn={isOwn}
   516→                      time={formatTime(message.createdAt)}
   517→                      loading={attachment.status === 'loading'}
   518→                    />
   519→                  )
   520→                ) : (
   521→                  <div className="text-sm whitespace-pre-wrap break-words">
   522→                    {message.content}
   523→                      </div>
   524→                    )}
   525→
   526→                {/* Meta */}
   527→                {!editing && !isVoice && !isSticker && !isImage && !isFile && (
   528→                  <div className={cn('flex items-center gap-1 justify-end mt-0.5 -mb-0.5 text-[10px]', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   529→                    {starred && <Star className="h-2.5 w-2.5 fill-amber-400 text-amber-400" />}
   530→                    {message.editedAt && <span>edited</span>}
   531→                    {message.expiresIn && <Clock className="h-2.5 w-2.5" />}
   532→                    <span>{formatTime(message.createdAt)}</span>

   589→                    className="h-9 w-9 rounded-lg hover:bg-accent hover:scale-110 transition-all flex items-center justify-center text-xl"
   590→                  >
   591→                    {e}
   592→                  </button>
   593→                ))}
   594→              </div>
   595→            </ContextMenuSubContent>
   596→          </ContextMenuSub>
   597→          {isOwn && !isDeleted && (
   598→            <>
   599→              <ContextMenuSeparator />
   600→              <ContextMenuItem onClick={() => { setEditing(true); setEditText(message.content) }}>
   601→                <Pencil className="h-4 w-4 mr-2" /> Edit
   602→              </ContextMenuItem>
   603→              <ContextMenuItem onClick={() => handleDelete(false)} className="text-muted-foreground">
   604→                <Trash2 className="h-4 w-4 mr-2" /> Delete for me
   605→              </ContextMenuItem>
   606→              <ContextMenuItem onClick={() => handleDelete(true)} className="text-destructive focus:text-destructive">
   607→                <Trash2 className="h-4 w-4 mr-2" /> Delete for everyone
   608→              </ContextMenuItem>
   609→            </>
   610→          )}
   611→        </ContextMenuContent>
   612→      </ContextMenu>
   613→
   614→      {forwardOpen && (
   615→        <Suspense fallback={null}>
   616→          <ForwardDialog
   617→            open={forwardOpen}
   618→            onOpenChange={setForwardOpen}
   619→            messageId={message.id}
   620→          />
   621→        </Suspense>
   622→      )}
   623→    </>
   624→  )

   627→export const MessageItem = memo(MessageItemImpl, (prev, next) => {
   628→  // Custom comparator: skip re-render when nothing visual changed.
   629→  // Compares message identity + the small set of fields that affect the
   630→  // rendered output. Avoids re-rendering the whole list when an unrelated
   631→  // store slice (e.g. typing indicators, online users) changes.
   632→  //
   633→  // Reactions and sender are compared by reference — the Zustand store always
   634→  // allocates new arrays/objects on updates (see updateMessage /
   635→  // updateReactionInStore), so a reference change reliably signals a real
   636→  // content change. This avoids the length-only pitfall (e.g. swap 👍 for ❤️
   637→  // keeps length identical but visually differs).
   638→  const pm = prev.message
   639→  const nm = next.message
   640→  return (
   641→    pm === nm ||
   642→    (pm.id === nm.id &&
   643→      pm.content === nm.content &&
   644→      pm.status === nm.status &&
   645→      pm.type === nm.type &&
   646→      pm.deletedAt === nm.deletedAt &&
   647→      pm.editedAt === nm.editedAt &&
   648→      pm.duration === nm.duration &&
   649→      pm.expiresIn === nm.expiresIn &&
   650→      pm.senderId === nm.senderId &&
   651→      pm.sender === nm.sender &&
   652→      pm.reactions === nm.reactions &&
   653→      pm.replyTo === nm.replyTo &&
   654→      prev.isOwn === next.isOwn &&
   655→      prev.isFirstInGroup === next.isFirstInGroup &&
   656→      prev.isLastInGroup === next.isLastInGroup &&
   657→      prev.showAvatar === next.showAvatar)
   658→  )
   659→})

   661→// Voice message bubble with waveform + play button
   662→function VoiceBubble({
   663→  duration,
   664→  playing,
   665→  progress,
   666→  onToggle,
   667→  isOwn,
   668→  time,
   669→  loading = false,
   670→}: {
   671→  duration: number
   672→  playing: boolean
   673→  progress: number
   674→  onToggle: () => void
   675→  isOwn: boolean
   676→  time: string
   677→  loading?: boolean
   678→}) {
   679→  const bars = Math.min(Math.max(Math.floor(duration * 2), 12), 40)
   680→  return (
   681→    <div className="flex items-center gap-2.5 min-w-[180px] py-0.5">
   682→      <button
   683→        onClick={onToggle}
   684→        disabled={loading}
   685→        className={cn(
   686→          'h-9 w-9 rounded-full flex items-center justify-center shrink-0 zc-tap',
   687→          isOwn ? 'bg-white/20' : 'bg-primary',
   688→          loading && 'opacity-60'
   689→        )}
   690→      >
   691→        {loading ? (
   692→          <Loader2 className="h-4 w-4 animate-spin" />
   693→        ) : playing ? (
   694→          <Pause className="h-4 w-4" />
   695→        ) : (
   696→          <Play className="h-4 w-4 ml-0.5" />
   697→        )}
   698→      </button>
   699→      <div className="flex-1">
   700→        <div className="flex items-center gap-[2px] h-6">
   701→          {Array.from({ length: bars }).map((_, i) => {
   702→            const filled = (i / bars) * 100 < progress
   703→            const h = 30 + Math.abs(Math.sin(i * 1.7)) * 70
   704→            return (
   705→              <div
   706→                key={i}
   707→                className={cn(
   708→                  'flex-1 rounded-full transition-colors',
   709→                  filled ? (isOwn ? 'bg-white' : 'bg-primary') : (isOwn ? 'bg-white/40' : 'bg-muted-foreground/40'),
   710→                  playing && filled && 'zc-wave-bar'
   711→                )}
   712→                style={{ height: `${h}%`, animationDelay: `${i * 0.05}s` }}
   713→              />
   714→            )
   715→          })}
   716→        </div>
   717→        <div className={cn('flex items-center justify-between text-[10px] mt-1', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   718→          <span>{loading ? '…' : playing ? `${Math.ceil(duration * (1 - progress / 100))}s` : `${duration}s`}</span>
   719→          <span>{time}</span>
   720→        </div>
   721→      </div>
   722→    </div>
   723→  )
   724→}

   726→// Delivery status ticks: ✓ sent, ✓✓ delivered, ✓✓ read (blue)
