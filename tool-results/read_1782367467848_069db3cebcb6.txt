   748‚Üí  /// Resolve what to show as the AppBar title for the current chat ‚Äî for
   749‚Üí  /// direct chats, the other member's display name; otherwise the chat title.
   750‚Üí  String _resolveTitle() {
   751‚Üí    final chat = widget.chat;
   752‚Üí    if (chat.type == 'saved') return 'Saved Messages';
   753‚Üí    if (chat.type == 'direct') {
   754‚Üí      final me = context.read<AuthService>().currentUser?.id;
   755‚Üí      final other = chat.members
   756‚Üí          .where((m) => m.user.id != me)
   757‚Üí          .firstOrNull;
   758‚Üí      return other?.user.name ?? chat.title;
   759‚Üí    }
   760‚Üí    return chat.title;
   761‚Üí  }
   762‚Üí
   763‚Üí  ({String emoji, String color}) _resolveAvatar() {
   764‚Üí    final chat = widget.chat;
   765‚Üí    if (chat.type == 'saved') {
   766‚Üí      return (emoji: 'bookmark', color: chat.avatarColor);
   767‚Üí    }
   768‚Üí    if (chat.type == 'direct') {
   769‚Üí      final me = context.read<AuthService>().currentUser?.id;
   770‚Üí      final other = chat.members
   771‚Üí          .where((m) => m.user.id != me)
   772‚Üí          .firstOrNull;
   773‚Üí      return (
   774‚Üí        emoji: other?.user.avatarEmoji ?? chat.avatarEmoji,
   775‚Üí        color: other?.user.avatarColor ?? chat.avatarColor,
   776‚Üí      );
   777‚Üí    }
   778‚Üí    return (emoji: chat.avatarEmoji, color: chat.avatarColor);
   779‚Üí  }

  1318‚Üí    final content = msg.content;

  1331‚Üí    if (content == _kDelivered || content.isEmpty) {
  1332‚Üí      _cache[msg.id] = _kDelivered;

  1336‚Üí          _loading = false;

  1339‚Üí      return;
  1340‚Üí    }

  1343‚Üí    if (content.startsWith('data:')) {

  1345‚Üí      if (mounted) {
  1346‚Üí        setState(() {
  1347‚Üí          _resolved = content;
  1348‚Üí          _loading = false;

  1350‚Üí      }
  1351‚Üí      return;

  1363‚Üí              _resolved = _kError;
  1364‚Üí              _loading = false;
  1365‚Üí            });
  1366‚Üí          }
  1367‚Üí          return;
  1368‚Üí        }

  1376‚Üí          });
  1377‚Üí        }
  1378‚Üí      } catch (_) {

  1380‚Üí        if (mounted) {
  1381‚Üí          setState(() {
  1382‚Üí            _resolved = _kError;
  1383‚Üí            _loading = false;
  1384‚Üí          });
  1385‚Üí        }
  1386‚Üí      }

  1388‚Üí    }

  1391‚Üí    _cache[msg.id] = content;
  1392‚Üí    if (mounted) {
  1393‚Üí      setState(() {
  1394‚Üí        _resolved = content;
  1395‚Üí        _loading = false;
  1396‚Üí      });
  1397‚Üí    }
