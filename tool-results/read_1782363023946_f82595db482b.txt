    47→const ForwardDialog = lazy(() => import('./forward-dialog').then(m => ({ default: m.ForwardDialog })))

    50→const QUICK_REACTIONS = ['👍', '❤️', '🔥', '😂', '😮', '🎉', '👏', '🙏']
    51→
    52→interface MessageItemProps {
    53→  message: MessageWithSender
    54→  isOwn: boolean
    55→  isFirstInGroup: boolean
    56→  isLastInGroup: boolean
    57→  showAvatar: boolean
    58→}
    59→
    60→function MessageItemImpl({ message, isOwn, isFirstInGroup, isLastInGroup }: MessageItemProps) {
    61→  const currentUser = useChatStore((s) => s.currentUser)
    62→  const activeChat = useChatStore((s) => s.activeChat)
    63→  const updateMessage = useChatStore((s) => s.updateMessage)
    64→  const removeMessage = useChatStore((s) => s.removeMessage)
    65→  const [editing, setEditing] = useState(false)
    66→  const [editText, setEditText] = useState(message.content)
    67→  const [starred, setStarred] = useState(false)
    68→  const [showHoverBar, setShowHoverBar] = useState(false)
    69→  const [showQuickReact, setShowQuickReact] = useState(false)
    70→  const [forwardOpen, setForwardOpen] = useState(false)
    71→  // voice playback
    72→  const [playing, setPlaying] = useState(false)
    73→  const [progress, setProgress] = useState(0)
    74→  const playTimer = useRef<ReturnType<typeof setInterval> | null>(null)
    75→
    76→  // Attachment resolution state — for image / file / voice messages the
    77→  // `content` field may be an encrypted URL (Supabase path), an encrypted
    78→  // data URL (dev fallback), or the literal "[delivered]" placeholder after
    79→  // the server wipes content post-delivery. We resolve it asynchronously.
    80→  const [attachment, setAttachment] = useState<{
    81→    status: 'loading' | 'ready' | 'delivered' | 'error'
    82→    dataUrl: string | null
    83→  }>({ status: 'loading', dataUrl: null })
    84→
    85→  const isDeleted = !!message.deletedAt
    86→  const isSystem = message.type === 'system'
    87→  const isVoice = message.type === 'voice'
    88→  const isSticker = message.type === 'sticker'
    89→  const isImage = message.type === 'image'
    90→  const isFile = message.type === 'file'
    91→
    92→  // Resolve attachment content for image / file / voice messages.
    93→  // Deps are intentionally limited to message.id + message.content (stable
    94→  // across re-renders of the memoized component) plus the chat type which
    95→  // is needed for E2EE decryption.
    96→  const chatType = activeChat?.type || 'direct'
    97→  useEffect(() => {
    98→    if (!isImage && !isFile && !isVoice) return
    99→    let cancelled = false
   100→
   101→    async function resolve() {
   102→      const raw = message.content
   103→      if (!raw) {
   104→        if (!cancelled) setAttachment({ status: 'error', dataUrl: null })
   105→        return
   106→      }
   107→      // Fast-path: server wiped content after delivery confirmation
   108→      if (raw === '[delivered]') {
   109→        if (!cancelled) setAttachment({ status: 'delivered', dataUrl: null })
   110→        return
   111→      }
   112→      try {
   113→        let resolved = raw
   114→        // If content isn't already a URL or data URL, it's an encrypted
   115→        // ciphertext JSON string — decrypt it first.
   116→        if (
   117→          !resolved.startsWith('http://') &&
   118→          !resolved.startsWith('https://') &&
   119→          !resolved.startsWith('data:')
   120→        ) {
   121→          const { decryptMessageForChat } = await import('@/lib/e2ee')
   122→          resolved = await decryptMessageForChat(resolved, message.chatId, chatType)
   123→        }
   124→
   125→        if (cancelled) return
   126→
   127→        if (resolved === '[delivered]') {
   128→          setAttachment({ status: 'delivered', dataUrl: null })
   129→          return
   130→        }
   131→        if (resolved.startsWith('http://') || resolved.startsWith('https://')) {
   132→          // Supabase-stored attachment — fetch ciphertext bytes & decrypt
   133→          const dataUrl = await fetchAndDecryptAttachment(resolved, message.chatId, chatType)
   134→          if (!cancelled) setAttachment({ status: 'ready', dataUrl })
   135→        } else if (resolved.startsWith('data:')) {
   136→          // Dev fallback — content is already the decrypted data URL
   137→          if (!cancelled) setAttachment({ status: 'ready', dataUrl: resolved })
   138→        } else {
   139→          // Legacy plaintext (no E2EE) — treat as data URL if it looks like one,
   140→          // otherwise mark as error.
   141→          if (!cancelled) setAttachment({ status: 'ready', dataUrl: resolved })
   142→        }
   143→      } catch {
   144→        if (!cancelled) setAttachment({ status: 'error', dataUrl: null })
   145→      }
   146→    }
   147→
   148→    resolve()
   149→    return () => { cancelled = true }
   150→  }, [message.id, message.content, message.chatId, chatType, isImage, isFile, isVoice])
   151→
   152→  // reaction grouped by emoji
   153→  const reactionGroups = message.reactions.reduce<Record<string, { count: number; mine: boolean }>>((acc, r) => {
   154→    if (!acc[r.emoji]) acc[r.emoji] = { count: 0, mine: false }
   155→    acc[r.emoji].count++
   156→    if (r.user.id === currentUser?.id) acc[r.emoji].mine = true
   157→    return acc
   158→  }, {})
   159→
   160→  async function handleReact(emoji: string) {
   161→    try {
   162→      const data = await toggleReaction(message.chatId, message.id, emoji)
   163→      getSocket()?.emit('reaction', {
   164→        chatId: message.chatId,
   165→        messageId: message.id,
   166→        emoji,
   167→        userId: currentUser?.id,
   168→        added: data.added,
   169→      })
   170→      // locally refresh this message reactions
   171→      const refreshed = await apiGet<{ messages: any[] }>(`/api/${message.chatId}/messages?limit=200`)
   172→      if (refreshed.messages) {
   173→        const m = refreshed.messages.find((x: any) => x.id === message.id)
   174→        if (m) updateMessage(message.chatId, m)
   175→      }
   176→      setShowQuickReact(false)
   177→    } catch (e) {
   178→      console.error(e)
   179→    }
   180→  }
   181→
   182→  async function handleEdit() {
   183→    if (!editText.trim()) return
   184→    try {
   185→      const data = await apiPatch<{ message: any }>(`/api/${message.chatId}/messages?messageId=${message.id}`, {
   186→        content: editText.trim(),
   187→      })
   188→      if (data.message) {
   189→        updateMessage(message.chatId, data.message)
   190→        getSocket()?.emit('message-update', { chatId: message.chatId, message: data.message, action: 'edit' })
   191→      }
   192→      setEditing(false)
   193→    } catch (e) {
   194→      console.error(e)
   195→    }
   196→  }
   197→
   198→  async function handleDelete(forEveryone: boolean = false) {
   199→    try {
   200→      await apiDelete(`/api/${message.chatId}/messages?messageId=${message.id}${forEveryone ? '&forEveryone=true' : ''}`)
   201→      removeMessage(message.chatId, message.id)
   202→      getSocket()?.emit('message-update', {
   203→        chatId: message.chatId,
   204→        message: { ...message, deletedAt: new Date().toISOString(), content: '🗑️ Message deleted' },
   205→        action: 'delete',
   206→      })
   207→      toast.success(forEveryone ? 'Deleted for everyone' : 'Message deleted')
   208→    } catch (e) {
   209→      console.error(e)
   210→    }
   211→  }
   212→
   213→  async function handleStar() {
   214→    try {
   215→      const data = await toggleStar(message.chatId, message.id)
   216→      setStarred(data.starred)
   217→      toast.success(data.starred ? 'Added to starred' : 'Removed from starred')
   218→    } catch (e) {
   219→      console.error(e)
   220→    }
   221→  }
   222→
   223→  function handleCopy() {
   224→    navigator.clipboard.writeText(message.content)
   225→    toast.success('Copied to clipboard')
   226→  }
   227→
   228→  function startReply() {
   229→    window.dispatchEvent(new CustomEvent('zc-reply', { detail: message }))
   230→  }
   231→
   232→  const audioRef = useRef<HTMLAudioElement | null>(null)
   233→
   234→  function togglePlay() {
   235→    if (playing) {
   236→      setPlaying(false)
   237→      audioRef.current?.pause()
   238→      return
   239→    }
   240→
   241→    const audioContent = attachment.status === 'ready' ? attachment.dataUrl : null
   242→
   243→    if (!audioContent) {
   244→      if (attachment.status === 'delivered') {
   245→        toast.error('Voice message no longer available')
   246→      } else if (attachment.status === 'error') {
   247→        toast.error('Could not load voice message')
   248→      } else {
   249→        toast.error('Voice message is still loading…')
   250→      }
   251→      return
   252→    }
   253→
   254→    if (!audioContent.startsWith('data:audio')) {
   255→      // legacy voice message (no actual audio) — simulate playback
   256→      setPlaying(true)
   257→      const duration = message.duration || 5
   258→      const step = 100 / (duration * 10)
   259→      playTimer.current = setInterval(() => {
   260→        setProgress((p) => {
   261→          const next = p + step
   262→          if (next >= 100) {
   263→            setPlaying(false)
   264→            if (playTimer.current) clearInterval(playTimer.current)
   265→            return 0
   266→          }
   267→          return next
   268→        })
   269→      }, 100)
   270→      return
   271→    }
   272→
   273→    const audio = new Audio(audioContent)
   274→    audioRef.current = audio
   275→    audio.onended = () => {
   276→      setPlaying(false)
   277→      setProgress(0)
   278→    }
   279→    audio.ontimeupdate = () => {
   280→      if (audio.duration) {
   281→        setProgress((audio.currentTime / audio.duration) * 100)
   282→      }
   283→    }
   284→    audio.play()
   285→    setPlaying(true)
   286→  }
   287→
   288→  useEffect(() => {
   289→    return () => {
   290→      audioRef.current?.pause()
   291→      if (playTimer.current) clearInterval(playTimer.current)
   292→    }
   293→  }, [])
   294→
   295→  if (isSystem) {
   296→    return (
   297→      <div className="flex justify-center my-2">
   298→        <span className="px-3 py-1 rounded-full bg-background/80 backdrop-blur text-xs text-muted-foreground border">
   299→          {message.content}
   300→        </span>
   301→      </div>
   302→    )
   303→  }
   304→
   305→  const HoverActions = (
   306→    <AnimatePresence>
   307→      {showHoverBar && !editing && !isDeleted && (
   308→        <motion.div
   309→          initial={{ opacity: 0, scale: 0.85 }}
   310→          animate={{ opacity: 1, scale: 1 }}
   311→          exit={{ opacity: 0, scale: 0.85 }}
   312→          transition={{ duration: 0.15 }}
   313→          className={cn(
   314→            'absolute top-1/2 -translate-y-1/2 flex items-center gap-0.5 bg-background/95 backdrop-blur rounded-full shadow-lg border p-0.5 z-20',
   315→            isOwn ? '-left-1' : '-right-1'
   316→          )}
   317→          onMouseLeave={() => setShowHoverBar(false)}
   318→        >
   319→          <button onClick={() => setShowQuickReact((v) => !v)} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="React">
   320→            <Smile className="h-4 w-4" />
   321→          </button>
   322→          <button onClick={startReply} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Reply">
   323→            <Reply className="h-4 w-4" />
   324→          </button>
   325→          <button onClick={() => setForwardOpen(true)} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Forward">
   326→            <Forward className="h-4 w-4" />
   327→          </button>
   328→          <button onClick={handleStar} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Star">
   329→            <Star className={cn('h-4 w-4', starred && 'fill-amber-400 text-amber-400')} />
   330→          </button>
   331→          {isOwn && (
   332→            <button onClick={() => { setEditing(true); setEditText(message.content) }} className="h-7 w-7 rounded-full hover:bg-accent flex items-center justify-center" title="Edit">
   333→              <Pencil className="h-4 w-4" />
   334→            </button>
   335→          )}
   336→        </motion.div>
   337→      )}
   338→    </AnimatePresence>
   339→  )
   340→
   341→  return (
   342→    <>
   343→      <ContextMenu>
   344→        <ContextMenuTrigger asChild>
   345→          <div
   346→            className={cn('group relative flex items-end gap-2 px-1', isOwn ? 'justify-end' : 'justify-start', isLastInGroup ? 'mb-2' : 'mb-0.5')}
   347→            onMouseEnter={() => setShowHoverBar(true)}
   348→            onMouseLeave={() => { setShowHoverBar(false); setShowQuickReact(false) }}
   349→          >
   350→            {!isOwn && (
   351→              <div className="w-8 shrink-0">
   352→                {isLastInGroup && (
   353→                  <ChatAvatar emoji={message.sender.avatarEmoji} color={message.sender.avatarColor} size="sm" userId={message.sender.id} />
   354→                )}
   355→              </div>
   356→            )}
   357→
   358→            <div className={cn('relative max-w-[85%] sm:max-w-[65%] min-w-0 flex flex-col', isOwn ? 'items-end' : 'items-start')}>
   359→              {!isOwn && isFirstInGroup && activeChat?.type !== 'direct' && activeChat?.type !== 'saved' && (
   360→                <span className="text-xs font-bold ml-1 mb-0.5 text-primary">
   361→                  {message.sender.name}
   362→                </span>
   363→              )}
   364→
   365→              {HoverActions}
   366→
   367→              {/* Quick react popover */}
   368→              <AnimatePresence>
   369→                {showQuickReact && (
   370→                  <motion.div
   371→                    initial={{ opacity: 0, y: 8, scale: 0.9 }}
   372→                    animate={{ opacity: 1, y: 0, scale: 1 }}
   373→                    exit={{ opacity: 0, y: 8, scale: 0.9 }}
   374→                    transition={{ type: 'spring', stiffness: 400, damping: 25 }}
   375→                    className={cn(
   376→                      'absolute -top-11 bg-background/95 backdrop-blur rounded-full shadow-lg border p-1 flex items-center gap-0.5 z-30',
   377→                      isOwn ? 'right-0' : 'left-0'
   378→                    )}
   379→                  >
   380→                    {QUICK_REACTIONS.map((e) => (
   381→                      <button
   382→                        key={e}
   383→                        onClick={() => handleReact(e)}
   384→                        className="h-8 w-8 rounded-full hover:bg-accent hover:scale-125 transition-all flex items-center justify-center text-lg"
   385→                      >
   386→                        {e}
   387→                      </button>
   388→                    ))}
   389→                  </motion.div>
   390→                )}
   391→              </AnimatePresence>
   392→
   393→              <motion.div
   394→                layout
   395→                initial={{ opacity: 0, y: 6, scale: 0.97 }}
   396→                animate={{ opacity: 1, y: 0, scale: 1 }}
   397→                transition={{ type: 'spring', stiffness: 380, damping: 28 }}
   398→                className={cn(
   399→                  'relative rounded-2xl px-3.5 py-2 shadow-sm',
   400→                  isOwn
   401→                    ? 'bg-gradient-to-br from-primary to-primary text-primary-foreground rounded-br-md'
   402→                    : 'bg-background border rounded-bl-md',
   403→                  isDeleted && 'opacity-60 italic',
   404→                  starred && 'ring-1 ring-amber-400/50'
   405→                )}
   406→              >
   407→                {/* Reply preview */}
   408→                {message.replyTo && (
   409→                  <div className={cn('mb-1.5 pl-2 border-l-2 text-xs', isOwn ? 'border-white/50' : 'border-primary')}>
   410→                    <div className={cn('font-semibold', isOwn ? 'text-white/90' : 'text-primary')}>
   411→                      {message.replyTo.senderName}
   412→                    </div>
   413→                    <div className={cn('truncate max-w-[200px]', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   414→                      {message.replyTo.content}
   415→                    </div>
   416→                  </div>
   417→                )}
   418→
   419→                {editing ? (
   420→                  <div className="min-w-[200px]">
   421→                    <textarea
   422→                      value={editText}
   423→                      onChange={(e) => setEditText(e.target.value)}
   424→                      className={cn(
   425→                        'w-full bg-transparent outline-none resize-none text-sm',
   426→                        isOwn ? 'placeholder-white/60' : ''
   427→                      )}
   428→                      rows={2}
   429→                      autoFocus
   430→                    />
   431→                    <div className="flex justify-end gap-1 mt-1">
   432→                      <Button size="sm" variant="ghost" className="h-7 text-xs" onClick={() => { setEditing(false); setEditText(message.content) }}>
   433→                        Cancel
   434→                      </Button>
   435→                      <Button size="sm" className="h-7 text-xs bg-emerald-600 hover:bg-emerald-700" onClick={handleEdit}>
   436→                        Save
   437→                      </Button>
   438→                    </div>
   439→                  </div>
   440→                ) : isSticker ? (
   441→                  isAnimatedSticker(message.content) ? (
   442→                    <Suspense fallback={<span className="text-6xl leading-none">⭐</span>}>
   443→                      <AnimatedStickerDisplay name={message.content} size={140} />
   444→                    </Suspense>
   445→                  ) : isLegacyEmoji(message.content) ? (
   446→                    <span className="text-6xl leading-none">{message.content}</span>
   447→                  ) : (
   448→                    <img
   449→                      src={stickerIconUrl(message.content)}
   450→                      alt={message.content}
   451→                      width={128}
   452→                      height={128}
   453→                      loading="lazy"
   454→                      className="object-contain"
   455→                    />
   456→                  )
   457→                ) : isImage ? (
   458→                  attachment.status === 'delivered' ? (
   459→                    <AttachmentPlaceholder text="Image no longer available (delivered & wiped)" />
   460→                  ) : attachment.status === 'loading' ? (
   461→                    <AttachmentLoading label="Loading image…" />
   462→                  ) : attachment.status === 'error' ? (
   463→                    <AttachmentPlaceholder text="Failed to load image" isError />
   464→                  ) : attachment.dataUrl && attachment.dataUrl.startsWith('data:image') ? (
   465→                    <img
   466→                      src={attachment.dataUrl}
   467→                      alt="shared"
   468→                      loading="lazy"
   469→                      className="rounded-lg max-w-[280px] max-h-[280px] object-contain cursor-pointer"
   470→                      onClick={() => attachment.dataUrl && window.open(attachment.dataUrl, '_blank')}
   471→                    />
   472→                  ) : (
   473→                    <AttachmentPlaceholder text="Unsupported image format" isError />
   474→                  )
   475→                ) : isFile ? (
   476→                  attachment.status === 'delivered' ? (
   477→                    <AttachmentPlaceholder text="File no longer available (delivered & wiped)" />
   478→                  ) : attachment.status === 'loading' ? (
   479→                    <AttachmentLoading label="Loading file…" />
   480→                  ) : attachment.status === 'error' ? (
   481→                    <AttachmentPlaceholder text="Failed to load file" isError />
   482→                  ) : (
   483→                    <div className="flex items-center gap-2 min-w-[200px]">
   484→                      <a
   485→                        href={attachment.dataUrl || '#'}
   486→                        download={attachment.dataUrl ? 'file' : undefined}
   487→                        className="flex items-center gap-2 px-3 py-2 rounded-lg bg-black/10 hover:bg-black/20 transition-colors"
   488→                      >
   489→                        <FileIcon className="h-5 w-5" />
   490→                        <span className="text-sm">Download file</span>
   491→                      </a>
   492→                    </div>
   493→                  )
   494→                ) : isVoice ? (
   495→                  attachment.status === 'delivered' ? (
   496→                    <AttachmentPlaceholder text="Voice message no longer available (delivered & wiped)" />
   497→                  ) : attachment.status === 'error' ? (
   498→                    <AttachmentPlaceholder text="Failed to load voice message" isError />
   499→                  ) : (
   500→                    <VoiceBubble
   501→                      duration={message.duration || 5}
   502→                      playing={playing}
   503→                      progress={progress}
   504→                      onToggle={togglePlay}
   505→                      isOwn={isOwn}
   506→                      time={formatTime(message.createdAt)}
   507→                      loading={attachment.status === 'loading'}
   508→                    />
   509→                  )
   510→                ) : (
   511→                  <div className="text-sm whitespace-pre-wrap break-words">
   512→                    {message.content}
   513→                      </div>
   514→                    )}
   515→
   516→                {/* Meta */}
   517→                {!editing && !isVoice && !isSticker && !isImage && !isFile && (
   518→                  <div className={cn('flex items-center gap-1 justify-end mt-0.5 -mb-0.5 text-[10px]', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   519→                    {starred && <Star className="h-2.5 w-2.5 fill-amber-400 text-amber-400" />}
   520→                    {message.editedAt && <span>edited</span>}
   521→                    {message.expiresIn && <Clock className="h-2.5 w-2.5" />}
   522→                    <span>{formatTime(message.createdAt)}</span>
   523→                    {isOwn && !isDeleted && <DeliveryTicks status={message.status || 'sent'} />}
   524→                  </div>
   525→                )}
   526→              </motion.div>
   527→
   528→              {/* Reactions */}
   529→              {Object.keys(reactionGroups).length > 0 && (
   530→                <div className={cn('flex flex-wrap gap-1 mt-1', isOwn ? 'justify-end' : 'justify-start')}>
   531→                  {Object.entries(reactionGroups).map(([emoji, info]) => (
   532→                    <motion.button
   533→                      key={emoji}
   534→                      initial={{ scale: 0 }}
   535→                      animate={{ scale: 1 }}

   602→      </ContextMenu>

   606→          <ForwardDialog
   607→            open={forwardOpen}

   609→            messageId={message.id}

   613→    </>

   617→export const MessageItem = memo(MessageItemImpl, (prev, next) => {
   618→  // Custom comparator: skip re-render when nothing visual changed.
   619→  // Compares message identity + the small set of fields that affect the
   620→  // rendered output. Avoids re-rendering the whole list when an unrelated
   621→  // store slice (e.g. typing indicators, online users) changes.
   622→  //
   623→  // Reactions and sender are compared by reference — the Zustand store always
   624→  // allocates new arrays/objects on updates (see updateMessage /
   625→  // updateReactionInStore), so a reference change reliably signals a real
   626→  // content change. This avoids the length-only pitfall (e.g. swap 👍 for ❤️
   627→  // keeps length identical but visually differs).
   628→  const pm = prev.message
   629→  const nm = next.message
   630→  return (
   631→    pm === nm ||
   632→    (pm.id === nm.id &&
   633→      pm.content === nm.content &&
   634→      pm.status === nm.status &&
   635→      pm.type === nm.type &&
   636→      pm.deletedAt === nm.deletedAt &&
   637→      pm.editedAt === nm.editedAt &&
   638→      pm.duration === nm.duration &&
   639→      pm.expiresIn === nm.expiresIn &&
   640→      pm.senderId === nm.senderId &&
   641→      pm.sender === nm.sender &&
   642→      pm.reactions === nm.reactions &&
   643→      pm.replyTo === nm.replyTo &&
   644→      prev.isOwn === next.isOwn &&
   645→      prev.isFirstInGroup === next.isFirstInGroup &&
   646→      prev.isLastInGroup === next.isLastInGroup &&
   647→      prev.showAvatar === next.showAvatar)
   648→  )
   649→})

   651→// Voice message bubble with waveform + play button
   652→function VoiceBubble({
   653→  duration,
   654→  playing,
   655→  progress,
   656→  onToggle,
   657→  isOwn,
   658→  time,
   659→  loading = false,
   660→}: {
   661→  duration: number
   662→  playing: boolean
   663→  progress: number
   664→  onToggle: () => void
   665→  isOwn: boolean
   666→  time: string
   667→  loading?: boolean
   668→}) {
   669→  const bars = Math.min(Math.max(Math.floor(duration * 2), 12), 40)
   670→  return (
   671→    <div className="flex items-center gap-2.5 min-w-[180px] py-0.5">
   672→      <button
   673→        onClick={onToggle}
   674→        disabled={loading}
   675→        className={cn(
   676→          'h-9 w-9 rounded-full flex items-center justify-center shrink-0 zc-tap',
   677→          isOwn ? 'bg-white/20' : 'bg-primary',
   678→          loading && 'opacity-60'
   679→        )}
   680→      >
   681→        {loading ? (
   682→          <Loader2 className="h-4 w-4 animate-spin" />
   683→        ) : playing ? (
   684→          <Pause className="h-4 w-4" />
   685→        ) : (
   686→          <Play className="h-4 w-4 ml-0.5" />
   687→        )}
   688→      </button>
   689→      <div className="flex-1">
   690→        <div className="flex items-center gap-[2px] h-6">
   691→          {Array.from({ length: bars }).map((_, i) => {
   692→            const filled = (i / bars) * 100 < progress
   693→            const h = 30 + Math.abs(Math.sin(i * 1.7)) * 70
   694→            return (
   695→              <div
   696→                key={i}
   697→                className={cn(
   698→                  'flex-1 rounded-full transition-colors',
   699→                  filled ? (isOwn ? 'bg-white' : 'bg-primary') : (isOwn ? 'bg-white/40' : 'bg-muted-foreground/40'),
   700→                  playing && filled && 'zc-wave-bar'
   701→                )}
   702→                style={{ height: `${h}%`, animationDelay: `${i * 0.05}s` }}
   703→              />
   704→            )
   705→          })}
   706→        </div>
   707→        <div className={cn('flex items-center justify-between text-[10px] mt-1', isOwn ? 'text-white/75' : 'text-muted-foreground')}>
   708→          <span>{loading ? '…' : playing ? `${Math.ceil(duration * (1 - progress / 100))}s` : `${duration}s`}</span>
   709→          <span>{time}</span>
   710→        </div>
   711→      </div>
   712→    </div>
   713→  )
   714→}
